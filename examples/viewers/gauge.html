function addViewers() {
    metricsViewer.addGauge('#gauge-div',
        "Gauge Metrics Viewer Example",
        "Metric Viewer Example of Gauge Metrics",
        "gauge.test.metric.a");
    }

$(function() {
    addViewers();
    metricsViewer.init();
    keepRefreshingViewer();
});

function keepRefreshingViewer() {
    refreshViewer();
    setInterval(refreshViewer, 2000);
}

function refreshViewer() {
    $.getJSON('data/gauge-metrics.json',
        function (data) {
            metricsViewer.refresh(data);
    });
};
