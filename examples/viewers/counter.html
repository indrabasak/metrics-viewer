function addViewers() {
    metricsViewer.addCounter('#counter-div',
    "Counter Metrics Viewer Example",
    "Metric Viewer Example of Counter Metrics",
    "counter.test.metric.a");
}

$(function() {
    addViewers();
    metricsViewer.init();
    keepRefreshingViewer();
});

function keepRefreshingViewer() {
    refreshViewer();
    setInterval(refreshViewer, 2000);
}

function refreshViewer() {
    $.getJSON('data/counter-metrics.json',
            function (data) {
        metricsViewer.refresh(data);
    });
}
