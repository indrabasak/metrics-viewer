function addViewers() {
    metricsViewer.addJvm('#jvm-div',
        "JVM Metrics Viewer Example",
        "Metric Viewer Example of JVM Metrics");
}

$(function() {
    addViewers();
    metricsViewer.init();
    keepRefreshingViewer();
});

function keepRefreshingViewer() {
    refreshViewer();
    setInterval(refreshViewer, 2000);
}

function refreshViewer() {
    $.getJSON('data/jvm-metrics.json',
            function (data) {
        metricsViewer.refresh(data);
    });
}
